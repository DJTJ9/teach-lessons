🚀 Lektion 08 — Praxisprojekt: Mini-Webhook-Server

Ziel: Alle Node.js-Kenntnisse in einem echten Webhook-Server zusammenführen der Telegram-ähnliche Nachrichten empfängt und verarbeitet.

Was bauen wir?

Ein Webhook-Server ist das Gegenteil von Polling: Statt dass dein Code regelmäßig fragt "Neue Nachrichten?", schickt Telegram die Nachrichten direkt an deinen Server wenn sie ankommen.

Unser Server empfängt: Simulated Telegram Webhook-Payload → verarbeitet Befehle → loggt Ergebnis

require('dotenv').config();

const config = {
    port: Number(process.env.PORT) || 3000,
    webhookSecret: process.env.WEBHOOK_SECRET,
};

if (!config.webhookSecret) {
    throw new Error('WEBHOOK_SECRET muss gesetzt sein');

src/handlers.js — Befehlslogik

// Verarbeitet Nachrichten und gibt Antwort zurück
async function verarbeiteNachricht(nachricht) {
    const text = (nachricht.text || '').trim().toLowerCase();
    const chatId = nachricht.chat?.id || 0;
    
    console.log(`[Chat ${chatId}] Empfangen: "${text}"`);
    
    // Befehlsverarbeitung:
    if (text === 'moin' || text === 'hallo') {
        return { chat_id: chatId, text: 'Moin! Wie kann ich helfen?' };
    }
    
    if (text.startsWith('task:')) {
        const taskText = text.slice(5).trim();
        if (!taskText) {
            return { chat_id: chatId, text: 'Was soll der Task heißen?' };
        }
        return { chat_id: chatId, text: `✅ Task erstellt: "${taskText}"` };
    }
    
    if (text === 'hilfe') {
        return {
            chat_id: chatId,
            text: 'Befehle:\nmoin — Begrüßung\ntask: [Text] — Task erstellen\nhilfe — diese Hilfe'
        };
    }
    
    return { chat_id: chatId, text: `Unbekannter Befehl: "${text}"` };
}

module.exports = { verarbeiteNachricht };

src/server.js — Hauptserver

const express = require('express');
const config = require('./config');
const { verarbeiteNachricht } = require('./handlers');

const app = express();
app.use(express.json());

// Logging Middleware:
app.use((req, res, next) => {
    console.log(`${new Date().toISOString()} ${req.method} ${req.url}`);
    next();
});

// Health Check — prüft ob der Server läuft:
app.get('/health', (req, res) => {
    res.json({ status: 'ok', zeit: new Date().toISOString() });
});

// Webhook-Endpunkt — hier kommen Telegram-Nachrichten an:
app.post('/webhook', async (req, res) => {
    // Secret prüfen (Sicherheit):
    const secret = req.headers['x-webhook-secret'];
    if (secret !== config.webhookSecret) {
        console.warn('Ungültiges Webhook-Secret!');
        return res.status(401).json({ fehler: 'Unauthorized' });
    }
    
    // Payload validieren:
    const { message } = req.body;
    if (!message) {
        return res.status(400).json({ fehler: 'message fehlt' });
    }
    
    try {
        // Nachricht verarbeiten:
        const antwort = await verarbeiteNachricht(message);
        console.log(`Antwort: "${antwort.text}"`);
        
        // 200 OK an Telegram zurück:
        res.json({ ok: true, antwort });
    } catch (fehler) {
        console.error('Verarbeitungsfehler:', fehler.message);
        res.status(500).json({ fehler: 'Interner Fehler' });
    }
});

// 404 Handler:
app.use((req, res) => {
    res.status(404).json({ fehler: 'Route nicht gefunden' });
});

app.listen(config.port, () => {
    console.log(`Webhook-Server läuft auf Port ${config.port}`);
    console.log(`Health: http://localhost:${config.port}/health`);
    console.log(`Webhook: POST http://localhost:${config.port}/webhook`);
});

# Task-Befehl:

  -H "Content-Type: application/json" \
  -H "x-webhook-secret: mein_geheimer_token" \
  -d '{"message": {"chat": {"id": 123}, "text": "task: Einkaufen"}}'

# Ohne Secret — sollte 401 geben:
curl -X POST http://localhost:3000/webhook \
  -H "Content-Type: application/json" \
  -d '{"message": {"text": "moin"}}'
Verbindung zum Chatbot: Dein Python-Bot nutzt Polling (bot.infinity_polling()). Der Webhook-Ansatz ist professioneller — Telegram schickt Nachrichten direkt statt dass der Bot fragt. Beide Ansätze nutzen dieselbe Telegram API, nur andersrum.

🧠 Erweiterungsaufgaben

1. Empfangene Nachrichten in eine JSON-Datei loggen
In handlers.js: const fs = require('fs/promises'); await fs.appendFile('log.json', JSON.stringify({...}) + '\n');
2. Rate-Limiting hinzufügen (max 10 Requests/Minute pro IP)
npm install express-rate-limit dann: const rateLimit = require('express-rate-limit'); app.use('/webhook', rateLimit({ windowMs: 60000, max: 10 }));
3. Antwort wirklich an Telegram senden
In handlers.js: await fetch(`https://api.telegram.org/bot${TOKEN}/sendMessage`, { method: 'POST', headers: {'Content-Type':'application/json'}, body: JSON.stringify(antwort) });
