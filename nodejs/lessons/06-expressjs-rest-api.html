🚀 Lektion 06 — Express.js: REST-API

Ziel: Express.js installieren und eine vollständige REST-API mit GET- und POST-Routen bauen.

Warum Express?

Das vanilla http-Modul (Lektion 5) funktioniert, ist aber repetitiv. Express.js ist das beliebteste Node.js-Framework und macht 80% des Boilerplate weg:

Vanilla http — POST-Body lesen

let body = '';
req.on('data', chunk => body += chunk);
req.on('end', () => {
    const daten = JSON.parse(body);
    // ... verarbeiten
});

Express — POST-Body lesen

// Express macht das automatisch:
app.post('/webhook', (req, res) => {
    const daten = req.body;  // fertig!
    // ... verarbeiten
});

Setup

mkdir meine-api
cd meine-api
npm init -y
npm install express

Erste Express-App

// app.js
const express = require('express');
const app = express();

// Middleware: JSON-Bodies automatisch parsen
app.use(express.json());

// GET Route:
app.get('/', (req, res) => {
    res.json({ nachricht: 'Willkommen bei meiner API!' });
});

// GET mit URL-Parameter:
app.get('/benutzer/:id', (req, res) => {
    const id = req.params.id;
    res.json({ benutzer_id: id, name: 'Tjark' });
});

// POST Route:
app.post('/todos', (req, res) => {
    const { text, prioritaet } = req.body;
    // normalerweise: in Datenbank speichern
    res.status(201).json({ 
        id: Date.now(), 
        text, 
        prioritaet,
        erstellt: new Date().toISOString() 
    });
});

// Server starten:
app.listen(3000, () => {
    console.log('API läuft auf http://localhost:3000');
});

Route-Parameter und Query-Strings

// URL-Parameter: /benutzer/42
app.get('/benutzer/:id', (req, res) => {
    const { id } = req.params;   // "42"
    res.json({ id });
});

// Query-String: /suche?q=hallo&limit=10
app.get('/suche', (req, res) => {
    const { q, limit = 10 } = req.query;  // "hallo", 10
    res.json({ suchbegriff: q, limit: Number(limit) });
});

// POST mit Body:
app.post('/nachrichten', (req, res) => {
    const { text, chat_id } = req.body;
    
    if (!text || !chat_id) {
        return res.status(400).json({ fehler: 'text und chat_id erforderlich' });
    }
    
    res.status(201).json({ gesendet: true, text });
});

Middleware — Code der zwischen Request und Route läuft

// Logging-Middleware (läuft vor jeder Route):
app.use((req, res, next) => {
    console.log(`${req.method} ${req.url} — ${new Date().toISOString()}`);
    next();  // nächste Middleware/Route aufrufen
});

// Auth-Middleware (nur für bestimmte Routen):
function pruefeToken(req, res, next) {
    const token = req.headers['x-api-token'];
    if (token !== process.env.API_TOKEN) {
        return res.status(401).json({ fehler: 'Unauthorized' });
    }
    next();
}

// Nur /admin Route schützen:
app.get('/admin', pruefeToken, (req, res) => {
    res.json({ daten: 'geheime Admindaten' });
});

Vollständige TODO-API

const express = require('express');
const app = express();
app.use(express.json());

let todos = [];  // Im echten Projekt: Datenbank
let nextId = 1;

app.get('/todos', (req, res) => {
    res.json(todos);
});

app.post('/todos', (req, res) => {
    const { text } = req.body;
    if (!text) return res.status(400).json({ fehler: 'text fehlt' });
    
    const todo = { id: nextId++, text, erledigt: false };
    todos.push(todo);
    res.status(201).json(todo);
});

app.put('/todos/:id/erledigt', (req, res) => {
    const todo = todos.find(t => t.id === Number(req.params.id));
    if (!todo) return res.status(404).json({ fehler: 'Todo nicht gefunden' });
    
    todo.erledigt = true;
    res.json(todo);
});

app.delete('/todos/:id', (req, res) => {
    const index = todos.findIndex(t => t.id === Number(req.params.id));
    if (index === -1) return res.status(404).json({ fehler: 'Nicht gefunden' });
    
    todos.splice(index, 1);
    res.json({ geloescht: true });
});

app.listen(3000, () => console.log('TODO-API läuft'));

🧠 Quiz

Frage 1: Was macht app.use(express.json())?

A) Nur JSON-Responses erlauben
❌ Falsch. Es parst eingehende JSON-Requests automatisch und macht den Body in req.body verfügbar.
B) JSON-Bodies automatisch parsen und in req.body verfügbar machen
✅ Richtig! Ohne diese Middleware ist req.body undefined.

Frage 2: Wie gibt man einen 404-Fehler mit JSON-Body zurück?

A) res.json({ fehler: 'nicht gefunden' })
❌ Fast. Das gibt 200 OK zurück, nicht 404. Status Code fehlt.
B) res.status(404).json({ fehler: 'nicht gefunden' })
✅ Richtig! .status() setzt den HTTP Status Code, .json() sendet die Antwort.

📋 Zusammenfassung

  • Express = Flask/FastAPI-Äquivalent für Node.js
  • app.use(express.json()): JSON-Bodies automatisch parsen
  • app.get/post/put/delete(route, handler): Routen definieren
  • req.params: URL-Parameter (/user/:id)
  • req.query: Query-String (?key=value)
  • req.body: POST/PUT Request-Body
  • res.status(code).json(obj): Antwort mit Status Code und JSON
  • Middleware: Code der vor Routes läuft (next() aufrufen!)
